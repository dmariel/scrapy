41 def _parse_sitemap(self, response): 

42 if response.url.endswith('/robots.txt'): 

43 for url in sitemap_urls_from_robots(response.text, base_url=response.url): 

44 yield Request(url, callback=self._parse_sitemap) 

45 else: 

46 body = self._get_sitemap_body(response) 

47 if body is None: 

48 logger.warning("Ignoring invalid sitemap: %(response)s", 

49 {'response': response}, extra={'spider': self}) 

50 return 

51 

52 s = Sitemap(body) 

53 it = self.sitemap_filter(s) 

54 

55 if s.type == 'sitemapindex': 

56 for loc in iterloc(it, self.sitemap_alternate_links): 

57 if any(x.search(loc) for x in self._follow): 

58 yield Request(loc, callback=self._parse_sitemap) 

59 elif s.type == 'urlset':

61 for r, c in self._cbs:

66 def _get_sitemap_body(self, response): 

67 """Return the sitemap body contained in the given response, 

68 or None if the response is not a sitemap. 

69 """ 

70 if isinstance(response, XmlResponse): 

71 return response.body 

72 elif gzip_magic_number(response): 

73 return gunzip(response.body) 

74 # actual gzipped sitemap files are decompressed above ; 

75 # if we are here (response body is not gzipped) 

76 # and have a response for .xml.gz, 

77 # it usually means that it was already gunzipped 

78 # by HttpCompression middleware, 

79 # the HTTP response being sent with "Content-Encoding: gzip" 

80 # without actually being a .xml.gz file in the first place, 

81 # merely XML gzip-compressed on the fly, 

82 # in other word, here, we have plain XML 

83 elif response.url.endswith('.xml') or response.url.endswith('.xml.gz'): 

84 return response.body
