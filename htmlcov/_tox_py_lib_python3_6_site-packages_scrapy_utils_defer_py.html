67def parallel(iterable, count, callable, *args, **named): 

68 """Execute a callable over the objects in the given iterable, in parallel, 

69 using no more than ``count`` concurrent calls. 

70 

71 Taken from: https://jcalderone.livejournal.com/24285.html 

72 """ 

73 coop = task.Cooperator() 

74 work = (callable(elem, *args, **named) for elem in iterable) 

75 return defer.DeferredList([coop.coiterate(work) for _ in range(count)])
