164def _request_deferred(request): 

165 """Wrap a request inside a Deferred. 

166 

167 This function is harmful, do not use it until you know what you are doing. 

168 

169 This returns a Deferred whose first pair of callbacks are the request 

170 callback and errback. The Deferred also triggers when the request 

171 callback/errback is executed (i.e. when the request is downloaded) 

172 

173 WARNING: Do not call request.replace() until after the deferred is called. 

174 """ 

175 request_callback = request.callback 

176 request_errback = request.errback 

177 

178 def _restore_callbacks(result): 

179 request.callback = request_callback 

180 request.errback = request_errback 

181 return result 

182 

183 d = defer.Deferred() 

184 d.addBoth(_restore_callbacks) 

185 if request.callback: 

186 d.addCallbacks(request.callback, request.errback) 

187 

188 request.callback, request.errback = d.callback, d.errback 

189 return d
